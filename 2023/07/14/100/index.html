无重复字符的最长子串

  • 双指针、哈希

  • O(n)

  •   class Solution {
  public:
      int lengthOfLongestSubstring(string s) {
          if(s.empty())  return 0;
          int len = 1;
          unordered_set<char> token;
          for (int i = 0, j = 0; i < s.length(); i ++) {
              while(token.count(s[i])){
                  token.erase(s[j++]);
              }
              token.insert(s[i]);
              len = max(len , i - j + 1);
          }  
          return len;    
      }
  };
